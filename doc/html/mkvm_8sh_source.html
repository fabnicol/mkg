25 ## @details Create file \b setup_network on success for debugging purposes
26 ## @retval Otherwise exit 1 on failure

29 setup_network() {
30 
31  [ -f setup_network ] && return
32  local res=0
33  if "${GUI}"
34  then
35  if "${CLONEZILLA_INSTALL}"
36  then

42  fi
43  else
44  if "${CLONEZILLA_INSTALL}"

46  dhclient
47  else
48 
49  # Workaround a VirtualBox bug, you need a keyboard input here of
50  # some sort. This is why we wait some time in mkgentoo.sh
51 
52  read -p "[MSG] Waiting for keyboard input..." input_str
53  echo "[MSG] Got input string: $input_str" | tee -a setup_network.log
54  echo "[INF] Running dhcpcd..." | tee -a setup_network.log
55  dhcpcd -HD $(ifconfig | cut -d' ' -f1 | head -n 1 | cut -d':' -f1)
56  fi
57  fi
58  res=$?
59  [ ${res} = 0 ] && touch setup_network || {
60  echo "[ERR] Could not fix internet access!" | tee -a setup_network.log
61  "${GUI}" && exit 1 || shutdown -h now
62  }

65 ## @fn partition()
66 ## @brief Create partition table, \b /dev/sda1 (bios_grub), \b /dev/sda2 (boot),
67 ## \b dev/sda3 (swap) and \b /dev/sda4 (system)
68 ## @details Create file \b partition. @n
69 ## On error, fill this file with successive exit codes of commands and exit.@n
70 ## On success, just create empty file.
71 ## @warning The VM needs time to recognize /dev/sda in some cases, for unclear
72 ## reasons.
73 ## This may be a kernel issue or a VirtualBox issue.
74 ## @bug Same issue with mkswap and swapon. Cleaning VBox config/settings,
75 ## syncing and a bit of sleep fixed these issues for the \e net-setup method.
76 ## @bug However if vm type is \e 'headless' the \e dhcpcd method is consistently
77 ## hampered by a VBox bug, which is tentatively circumvented
78 ## by sending a `controlvm keyboardputscancode 1c` instruction.
79 ## Tests show that this is linked to a requested user keyboard or mouse input by
80 ## the Gentoo minimal install CD. This cannot be simulated
81 ## owing to the lack of /dev/uinput. The reason why user input is requested has
82 ## not been found. Without it, /dev/sda2 and/or sda4 are
83 ## mistakenly identified as being mounted and/or busy, while this cannot be the
84 ## case. With even a single keystroke for a `read`command, all
85 ## falls back into place. This is why using the net-setup script, which forces
86 ## user input, circumvents the issue.
87 ## This may be caused by an aging kernel and/or incompatibilities with
88 ## virtualization.
89 ## Using a CloneZilla CD as a replacement solved the issue completely.
90 ## It might be better to use a beefed-up Gentoo install CD.
91 ## @note It might be necessary with older machines to increase the amount of
92 ## sleep.
93 ## @ingroup mkFileSystem
94 
95 partition() {
96 
97  [ -f partition ] && [ ! -s partition ] && return 0
98  sleep 5
99  parted --script \
100  --align=opt \
101  /dev/sda \
102  "mklabel gpt \
103  unit mib \
104  mkpart primary 1 3 \
105  name 1 grub \
106  set 1 bios_grub on \
107  mkpart primary 3 131\
108  name 2 boot \
109  mkpart primary 131 643 \
110  name 3 swap \
111  mkpart primary 643 -1 \
112  set 2 boot on"
113  res0=$?
114  sync
115  sleep 10
116  mkfs.fat -v -F 32 /dev/sda2
117  res1=$?
118  sync
119  sleep 10
120  mkfs.ext4 -e continue -q -F -F /dev/sda4
121  res2=$?
122  swapoff -a
123  sync
124  sleep 10
125  mkswap /dev/sda3
126  res3=$?
127  sync
128  swapon /dev/sda3
129  res4=$?
130  sync
131  findmnt /dev/sda4 && umount -l /dev/sda4
132  mount /dev/sda4 /mnt/gentoo
133  res5=$?
134  res=$((${res0} | ${res1} | ${res2} | ${res3} | ${res4} | ${res5}))
135  if [ ${res} = 0 ]
136  then
137  touch partition
138  echo "[MSG] Partioned /dev/sda4 correctly."
139  else
140  echo "[ERR] parted exit code: ${res0}" | tee partition.log
141  echo "[ERR] mkfs.fat exit code: ${res1}" | tee -a partition.log
142  echo "[ERR] mkfs.ext4 exit code: ${res2}" | tee -a partition.log
143  echo "[ERR] mkswap exit code: ${res3}" | tee -a partition.log
144  echo "[ERR] swapon exit code: ${res4}" | tee -a partition.log
145  echo "[ERR] mount exit code: ${res5}" | tee -a partition.log
146  echo "[ERR] Failed to cleanly partition main disk" | tee -a partition.log
147  sleep 10
148  if [ $((${res1} | ${res2} | ${res3} | ${res5})) != 0 ]
149  then
150  echo "[ERR] Critical errors while partitioning" | tee -a partition.log
151  swapoff -a
152  findmnt /dev/sda4 && umount -l /dev/sda4
153  if "${GUI}"
154  then
155  return 1
156  else
157  echo "[ERR] Shutting down in 5 seconds..."
158  sleep 5
159  shutdown -h now
160  fi
161  else
162  echo "[WAR] Parted issue but mkfs and mount OK. Going on..." \
163  | tee partition.log
164  return 1
165  fi
166  fi
167 }
