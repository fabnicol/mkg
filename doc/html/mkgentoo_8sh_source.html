277 ## @fn get_options()
278 ## @brief Parse command line
279 ## @ingroup createInstaller
280 
281 get_options() {
282 
283  local LOGGING="true"
284  local LOGGER0="$(which logger)"
285  local ECHO="$(which echo)"
286  [ -n "${LOGGER0}" ] && LOGGER_VERBOSE_OPTION="-s" || LOGGER0="${ECHO}"
287  LOG=("${LOGGER0}" "${LOGGER_VERBOSE_OPTION}")
288  export LOG
289 
290  if [ $# = 0 ]
291  then
292  ISO_OUTPUT="gentoo.iso"
293  CREATE_ISO=true
294  return 0
295  fi
296 
297  while (( "$#" ))
298  do
299  if grep -q '=' <<< "$1"
300  then
301  left=$(sed -E 's/([^=]*)=(.*)/\1/' <<< "$1")
302  right=$(sed -E 's/([^=]*)=(.*)$/\2/' <<< "$1")
303  if validate_option "${left}"
304  then
305  declare -u VAR=${left}
306  eval "${VAR}"=\"${right}\"
307  if [ "${VERBOSE}" = "true" ]
308  then
309  ${LOG[*]} "[CLI] Assign: ${VAR}=${!VAR}"
310  fi
311  fi
312  else
313  if grep -q "\.iso" <<< "$1"
314  then
315  ISO_OUTPUT="$1"
316  CREATE_ISO=true
317  else
318  if validate_option "$1"
319  then
320  declare -u VAR="$1"
321  eval "${VAR}"=true
322  [ "${VERBOSE}" = "true" ] \
323  && ${LOG[*]} "[CLI] Assign: ${VAR}=true"
324  fi
325  fi
326  fi
327  if [ "${LOGGING}" = "true" ]
328  then
329  LOGGER="${LOGGER0}"
330  else
331  LOGGER="${ECHO}"
332  LOGGER_VERBOSE_OPTION=""
333  fi
334 
335  # fallback
336 
337  [ -z "${LOGGER}" ] && LOGGER="${ECHO}"
338  if [ "${QUIET_MODE}" = "true" ]
339  then
340  LOGGER_VERBOSE_OPTION=""
341  fi
342 
343  LOG=("${LOGGER}" "${LOGGER_VERBOSE_OPTION}")
344  shift
345  done
346 }

348 ## @fn test_cli_pre()
349 ## @brief Check VirtualBox version and prepare commandline analysis

353 test_cli_pre() {
354 
355  { [ "${VERBOSE}" = "true" ] || [ "${DEBUG_MODE}" = "true" ]; } \
356  && [ "${QUIET_MODE}" = "true" ] \
357  && { ${LOG[*]} "[ERR] You cannot have 'quiet' and verbose modes at \

359  exit 1; }
360

362 
363  [ "$(whoami)" != "root" ] && { ${LOG[*]} "[ERR] must be root to continue"

365  [ -L /home/partimag ] && rm /home/partimag

413  <<< "${vbox_version}")

417  ${LOG[*]} "[ERR] VirtualBox must be at least version 6.1.10"
418  ${LOG[*]} "[ERR] Please update and reinstall"
419  do_exit=true

423

427

431  then
432 
433  # effectively correcting first pass assignment.
434  # This can be done only when cl is entirely parsed.
435 
436  CREATE_ISO="false"
437  fi

439  then
440  if [ "${VERBOSE}" = "true" ]
441  then
442  ${LOG[*]} "[MSG] Bootable ISO output is: ${ISO_OUTPUT}"

444  then
445  ${LOG[*]} "[MSG] Build Gentoo to bootable ISO"

448  fi
449  fi
450  fi
451

458 ## @fn test_cli()
459 ## @brief Analyse commandline
460 ## @param cli Commandline
461 ## @details @li Create globals of the form VAR=arg when there is var=arg on
462 ## commandline
463 ## @li Otherwise assign default values VAR=defaults (3rd argument in array #ARR)
464 ## @li Also checks type of argument against types described for #ARR
465 ## @ingroup createInstaller
466 
467 test_cli() {
468 
469  declare -i i=$1
470  local sw=${ARR[i*4]}
471  local desc=${ARR[i*4+1]}
472  local default0="${ARR[i*4+2]}"
473  eval default=\""${default0}"\"
474  local type=${ARR[i*4+3]}
475  local cli_arg=false
476  declare -u V=${sw}
477  local y=$(sed 's/.*://' <<< "${type}")
478  local cond0="${y^^}"
479  local cond
480  [ -n "${cond0}" ] && cond="${!cond0}"
481 
482  # Do not use ${} directly as true/false without [...] in
483  # this function and the above.
484  # as Boolean variables may not all be set yet.
485 
486  if [ -n "${!V}" ]
487  then
488  [ "${DEBUG_MODE}" = "true" ] \
489  && ${LOG[*]} "[CLI] ${desc}=${!V}"
490 
491  # checking on types among values found on commandline
492 
493  case "${type}" in
494  b) if [ "${!V}" != "true" ] && [ "${!V}" != "false" ]
495  then
496  ${LOG[*]} "[ERR] ${sw} is Boolean: specify its value as \
497 either 'false' or 'true' on command line"
498  exit 1
499  fi
500  ;;
501  d) [ ! -d "${!V}" ] \
502  && { ${LOG[*]} "[ERR] ${sw}=... must be an existing \
503 directory."
504  exit 1; }
505  ;;
506  e) if ! grep -q -E "[a-z]+@[a-z]+\.[a-z]+" <<< "${!V}"
507  then
508  ${LOG[*]} "[ERR] ${sw}=... must be a valid email \
509 address"
510  exit 1
511  fi
512  ;;
513  f) [ ! -f "${!V}" ] \
514  && { ${LOG[*]} "[ERR] ${sw}=... must be an existing file."
515  exit 1;} ;;
516  n) if ! test_numeric "${!V}"
517  then
518  ${LOG[*]} "[ERR] ${sw}=... is not numeric."
519  exit 1
520  fi
521  ;;
522  o) [ "${!V}" != "on" ] && [ "${!V}" != "off" ] \
523  && { ${LOG[*]} "[ERR] ${sw}=on or ${sw}=off are the only \
524  two possible values."
525  exit 1; }
526  ;;
527  u) if ! test_URL "${!V}"
528  then
529  ${LOG[*]} "[ERR] ${sw}=... must be a valid URL"
530  exit 1
531  fi
532  ;;
533  vm) [ "${VM}" = "false" ] && VM=""
534  ;;
535 
536  # conditional types of the form e/f/s:...
537 
538  *:*)
539  if [ "${cond}" != "true" ] \
540  && { [ "${cond}" = "false" ] || [ -z "${cond}" ]; }
541  then
542  if [ -z "${cond}" ]
543  then
544  ${LOG[*]} "[ERR] Execution cannot proceed without \
545 specifying an explicit value for ${y}=..."
546  else
547  ${LOG[*]} "[ERR] Execution cannot proceed as option \
548 values for ${y}=false and ${sw}=${!V} are incompatible."
549  fi
550  ${LOG[*]} "[ERR] Fatal. Exiting..."
551  exit 1
552  fi
553 
554  # [ -z "${!V}" ] <=> { [ "${cond}" = "false" ]
555  # || [ -z "${cond}" ]; } && [ "${cond}" != "true" ]
556 
557  esac
558  else
559  if [ "${cond}" = "true" ] || { [ "${cond}" != "false" ] \
560  && [ -n "${cond}" ]; }
561  then
562  local EXCEPT="FROM_${y^^}"
563  if [ -z "${default}" ] && ! "${!EXCEPT}"
564  then
565  ${LOG[*]} "[ERR] Execution cannot proceed without an explicit \
566  value for ${sw}=... as ${y}=${cond}"
567  ${LOG[*]} "[ERR] Fatal. Exiting..."
568  exit 1
569  fi
570  fi
571 
572  # not found on command line or erroneously empty
573  # replacing by default in any case, except if type == "s"
574  # and default empty. This is the case e.g. for passwds.
575 
576  if [ "${type}" = "s" ] && [ -z "${default}" ] && [ "${sw}" != "dep" ]
577  then
578  ${LOG[*]} "[ERR] Execution cannot proceed without explicit value \
579 for ${sw}"
580  if [ "${INTERACTIVE}" = "true" ]
581  then
582  local reply=""
583  while [ -z ${reply} ]
584  do
585  ${LOG[*]} "[MSG] Please enter value: "
586  read reply
587  eval "${V}"=\"${reply}\"
588  done
589  else
590  ${LOG[*]} "[ERR] Fatal. Exiting..."
591  exit 1
592  fi
593  fi
594  [ "${DEBUG_MODE}" = "true" ] \
595  && ${LOG[*]} "[CLI] Desc/default: ${desc}=${default}"
596  eval "${V}"=\""${default}"\"
597  fi
598 
599  # Post processing of arguments in list form [a,b...]
600 
601  if [ "${!V::1}" = "[" ]
602  then
603  local w="${!V}"
604  local V1="${w:1:(${#w}-2)}"
605  V1=$(sed 's/,/ /g' <<< ${V1})
606  eval "${V}"=\"${V1}\"
607  fi
608 
609  [ "${DEBUG_MODE}" = "true" ] && ${LOG[*]} "[MSG] Export: ${V}=\"${!V}\""
610 
611  # exporting is made necessary by usage in companion scripts.
612 
613  export "${V}"
614 }

1782 ## @fn mount_vdi()
1783 ## @brief Use qemu to mount VDI file to host folder
1784 ## @param "w" for enabling read-write mode, otherwise read-only.
1785 ## @warning May involve security issues, especially if "w" is enabled.
1786 ## @note Uses global variable ${SHARED_ROOT_DIR}
1787 ## @see Guestfish and qemu websites for security issues.
1788 
1789 mount_vdi() {
1790 
1791  cd "${VMPATH}" || exit 2
1792  if ! [ -d "${SHARED_ROOT_DIR}" ]
1793  then
1794  logger -s "[ERR] \"${SHARED_ROOT_DIR}\" is not a directory."
1795  exit 1
1796  fi
1797 
1798  if [ "$1" = "w" ]
1799  then
1800  logger -s "[WAR] Enabling write mode for VDI mount."
1801  logger -s "[WAR] This may cause security issues: take care of I/O and"
1802  logger -s "[WAR] networking security."
1803  fi
1804 
1805  check_tool qemu-nbd
1806  QEMU_NBD_BINARY=$(which qemu-nbd)
1807  modprobe nbd
1808  if_fails $? \
1809  "[ERR] Your linux kernel does not support the NBD protocol." \
1810  "[ERR] Please revise and rebuild your kernel configuration \
1811 so that **modprobe nbd** succeeds."
1812 
1813  declare -i j=1
1814 
1815  while [ $j -le 10 ]
1816  do
1817  if [ "$1" = "w" ]
1818  then
1819  "${QEMU_NBD_BINARY}" -c /dev/nbd${j} -f vdi "${VM}.vdi"
1820  else
1821  echo "j=$j"
1822  "${QEMU_NBD_BINARY}" --read-only -c /dev/nbd${j} -f vdi "${VM}.vdi"
1823  fi
1824  local res=$?
1825  if [ $res = 0 ]
1826  then
1827  echo "[MSG] Connected /dev/nbd${j}"
1828  else
1829  echo "[WAR] Could not connect VDI disk, qemu exit code: $res"
1830  echo " looping nbd${j}..."
1831  j=j+1
1832  continue
1833  fi
1834 
1835  sync
1836  sleep 2
1837  mount /dev/nbd${j}p4 "${SHARED_ROOT_DIR}"
1838 
1839  if [ $? != 0 ]
1840  then
1841  echo "[WAR] Failed to mount virtual disk ${VM}.vdi root \
1842 to ${SHARED_ROOT_DIR}."
1843  echo " Looping nbd${j}..."
1844  j=j+1
1845  "${QEMU_NBD_BINARY}" -d /dev/nbd${j}
1846  continue
1847  fi
1848 
1849  mount /dev/nbd${j}p2 "${SHARED_ROOT_DIR}/boot"
1850 
1851  if [ $? != 0 ]
1852  then
1853  echo "[WAR] Failed to mount virtual disk ${VM}.vdi kernel \
1854 to ${SHARED_ROOT_DIR} boot directory."
1855  echo " Looping nbd${j}..."
