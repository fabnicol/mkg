40 adjust_environment() {
41 
42  # Adjusting /etc/fstab
43 
44  local uuid2=$(blkid | grep sda2 | cut -f2 -d' ')
45  local uuid3=$(blkid | grep sda3 | cut -f2 -d' ')
46  local uuid4=$(blkid | grep sda4 | cut -f2 -d' ')
47  echo "Partition /dev/sda2 with ${uuid2}" | tee partition_log
48  echo "Partition /dev/sda3 with ${uuid3}" | tee -a partition_log
49  echo "Partition /dev/sda4 with ${uuid4}" | tee -a partition_log
50  echo "${uuid2} /boot vfat defaults 0 2" \
51  > /etc/fstab
52  echo "${uuid3} none swap sw 0 0" \
53  >> /etc/fstab
54  echo "${uuid4} / ext4 defaults 0 1" \
55  >> /etc/fstab
56  echo "/dev/cdrom /mnt/cdrom auto noauto,user,discard 0 0" \
57  >> /etc/fstab
58 
59  source /etc/profile
60 
61  # Refresh and rebuild @world frequently emerge complains about
62  # having to be upgraded before anything else. We shall use
63  # emerge-webrsync as emerge --sync is a bit less robust (rsync
64  # rotation bans...)
65 
66  if ! emerge-webrsync
67  then
68  echo "[ERR] emerge-webrsync failed!" | tee emerge.build
69  return 1
70  fi
71 
72  # One needs to build cmake without the qt5 USE value first,
73  # otherwise dependencies cannot be resolved.
74 
75  USE='-qt5' emerge -1 cmake
76  if [ $? != 0 ]
77  then
78  echo "emerge cmake failed!" | tee -a emerge.build
79  return 1
80  fi
81 
82  # There is an intractable circular dependency that
83  # can be broken by pre-emerging python
84 
85  USE="-sqlite -bluetooth" emerge -1 dev-lang/python | tee -a emerge.build
86  if [ $? != 0 ]
87  then
88  echo "emerge python failed!" | tee -a emerge.build
89  return 1
90  fi
91 
92  emerge -1 -u sys-app/portage
93 
94  # select profile (most recent plasma desktop)
95 
96  local profile=$(eselect --color=no --brief profile list \
97  | grep desktop \
98  | grep plasma \
99  | grep ${PROCESSOR} \
100  | grep -v systemd \
101  | head -n 1)
102 
103  eselect profile set ${profile}
104 
105  # Use and keywords (mkdir -p to neutralize error msg)
106 
107  mkdir -p /etc/portage/package.accept_keywords
108  mkdir -p /etc/portage/package.use
109  cp -vf "${ELIST}.accept_keywords" \
110  /etc/portage/package.accept_keywords/ \
111  | tee emerge.build
112 
113  cp -vf "${ELIST}.use" /etc/portage/package.use/ \
114  | tee emerge.build
115 
116  cp -vf "${ELIST}.accept_keywords" \
117  /etc/portage/package.accept_keywords/ \
118  | tee emerge.build
119 
120  # add logger. However it will not be usable for now,
121  # this is why we are using custom logs and tee's.
122 
123  emerge -uD app-admin/sysklogd
124  rc-update add sysklogd default
125  rc-service sysklogd start
126 
127  # other core sysapps to be merged first. LZ4 is a kernel
128  # dependency for newer linux kernels.
129 
130  emerge -u net/misc/wget
131  emerge -u app-arch/lz4 net-misc/netifrc sys-apps/pcmciautils
132 
133  if [ $? != 0 ]
134  then
135  echo "[ERR] emerge netifrs/pcmiautils failed!" | tee -a emerge.build
136  return 1
137  fi
138 
139  # Now on to updating @world set. Be patient and wait for about
140  # 15-24 hours
141  # as syslogd is not yet there we tee a custom build log
142  # emerging gcc and glibc is mainly for CFLAGS changes.
143 
144  emerge sys-devel/gcc
145  emerge sys-libs/glibc
146  ## ---- PATCH ----
147  #
148  # This is temporarily necessary while display-manager is not
149  # stabilized in the portage tree (March 2021)
150  # NOTE: should be retrieved later on
151 
152  emerge -q --unmerge sys-apps/sysvinit | tee -a emerge.build
153 
154  ## ---- End of patch ----
155 
156  emerge -uDN --with-bdeps=y @world 2>&1 | tee -a emerge.build
157  [ $? != 0 ] && {
158  echo "[ERR] emerge @world failed!" | tee -a emerge.build
159  return 1; }
160 
161  emerge -q -u sys-apps/sysvinit
162 
163  # Networking in the new environment
164 
165  echo hostname=${NONROOT_USER}pc > /etc/conf.d/hostname
166  cd /etc/init.d || exit 2
167  ln -s net.lo net.${iface}
168  cd - || exit 2
169  rc-update add net.${iface} default
170 
171  # Set keymaps and time
172 
173  if [ -n "${VM_LANGUAGE}" ]
174  then
175  echo "keymap=${VM_LANGUAGE}" > /etc/conf.d/keymaps
176  echo 'keymap="us"' >> /etc/conf.d/keymaps
177  else
178  echo 'keymap="us"' > /etc/conf.d/keymaps
179  fi
180  sed -i 's/clock=.*/clock="local"/' /etc/conf.d/hwclock
181  echo "${TIMEZONE}" > /etc/timezone
182  emerge -u --config sys-libs/timezone-data | tee -a emerge.build
183 
184  # Localization.
185 
186  echo "fr_FR.UTF-8 UTF-8" > /etc/locale.gen
187  echo "fr_FR ISO-8859-15" >> /etc/locale.gen
188  echo "en_US.UTF-8 UTF-8" >> /etc/locale.gen
189  echo "en_US ISO-8859-1" >> /etc/locale.gen
190  locale-gen | tee -a emerge.build
191  eselect locale set 1
192  env-update && source /etc/profile && export PS1="(chroot) ${PS1}"
193 }

195 ## @fn build_kernel()
196 ## @details
197 ## @li Emerge \b gentoo-sources, \b genkernel, \b pciutils and
198 ## \b linux-firmware
199 ## @li Mount /dev/sda2 to /boot/
200 ## @li Build kernel and initramfs. Log into kernel.log.
201 ## @retval -1 on error.
202 ## @ingroup mkFileSystem
203 
204 build_kernel() {
205 
206  # Building the kernel
207 
208  emerge gentoo-sources sys-kernel/genkernel pciutils \
209  | tee kernel.log
210 
211  # Now mount the new boot partition
212 
213  mount /dev/sda2 /boot
214  cp -vf .config /usr/src/linux
215  cd /usr/src/linux
216 
217  if [ "$PWD" != "/usr/src/linux" ]
218  then
219  echo "[ERR] Could not cd to /usr/src/linux" | tee -a kernel.log
220  exit 2
221  fi
222 
223  # kernel config issue here
224 
225  make syncconfig # replaces silentoldconfig as of 4.19
226  make -s -j${NCPUS} 2>&1 | tee -a kernel.log
227  make modules_install 2>&1 | tee -a kernel.log
228  make install 2>&1 | tee -a kernel.log
229  if [ $? != 0 ]
230  then
231  echo "[ERR] Kernel building failed!" | tee -a kernel.log
232  return 1
233  fi
234 
235  genkernel --install initramfs | tee -a kernel.log
236  emerge sys-kernel/linux-firmware | tee -a kernel.log
237  make clean
238 
239  if [ -f /boot/vmlinuz* ]
240  then
241  echo "[MSG] Kernel was built" | tee -a kernel.log
242  else
243  echo "[ERR] Kernel compilation failed!" | tee -a kernel.log
244  return 1
245  fi
246 
247  if [ -f /boot/initr*.img ]
248  then
249  echo "[MSG] initramfs was built" | tee -a kernel.log
250  else
251  echo "[ERR] initramfs compilation failed!" | tee -a kernel.log
252  return 1
253  fi
254 }

275 
276  local packages=`grep -E -v '(^\s*$|^\s*#.*$)' "${ELIST}"`
277 
278  # Trace for debugging
279 
280  echo "${packages}" > package_list
281 
282  # v1.3: adding --keep-going. Limited emerge failures may not render
283  # the build useless.
284  # allow some tolerance for incomplete builds caused
285  # by changes in portage tree package names and versions.
286 
287  # do not quote `packages' variable!
288 
289  emerge -uDN --with-bdeps=y --keep-going ${packages} 2>&1 \
290  | tee -a log_install_software.log
291  local res_install=$?
292 
293  if [ "${res_install}" != "0" ]
294  then
295  # one more chance, who knows
296  emerge --resume | tee -a log_install_software.log
297  res_install=$?
298  res_install=$?
299  fi
300 
301  if [ $? != 0 ]
302  then
303  # one more chance, who knows
304  emerge --resume | tee -a log_install_software.log
305  fi
306 
307  # do not use \ to continue line below:
308 
309  if ! "${MINIMAL}"
310  then
311  # this test is for second-chance debugging runs
312 
313  if ! Rscript -e "library(ggplot2)"
314  then
315  echo "install.packages(c('data.table', 'dplyr', 'ggplot2',
316 'bit64', 'devtools', 'rmarkdown'), repos=\"${CRAN_REPOS}\")" \
317  > libs.R
318  Rscript libs.R 2>&1 | tee Rlibs.log
319  rm -f libs.R
320  fi
321  fi
322 
323  # update environment
324 
325  env-update
326  source /etc/profile
327 
328  if [ "${res_install}" != "0" ]
329  then
330  echo "[ERR] Main package build step failed" \
331  | tee -a log_install_software.log
332  return 1
333  fi

361

363

365 
366  rc-update add cronie default
367  rc-update add display-manager default
368  rc-update add dbus default
369  rc-update add elogind boot
370

372 
373  for x in /etc/runlevels/default/net.*
374  do
375  rc-update del $(basename $x) default
376  rc-service --ifstarted $(basename $x) stop
377  done
378  rc-update del dhcpcd default
379  rc-update add NetworkManager default
380 
381  #--- groups and sudo
382 
383  useradd -m -G users,wheel,audio,video,plugdev \
384  -s /bin/bash "${NONROOT_USER}"
385

389  fi

392

394

396

400  return 3
401  fi
402

404 
405  if [ "${BIOS}" = "true" ]

408  else

410  fi
411

415  exit 1
416  fi
417

419 
420  if [ $? != 0 ]
421  then
422  echo "[ERR] Could not configure grub" | tee -a grub.log
423  exit 1
424  fi
