24 ## @fn fetch_gentoo_install_iso()
25 ## @brief Download minimal Gentoo install ISO file and caches.q
26 ## @details Optionaly checks MD5SUMS.
27 ## @ingroup fetchFunctions

29 fetch_gentoo_install_iso() {
30 
31  local verb="-v"
32  ! ${VERBOSE} && verb=""
33 
34  local GENTOO_LIST="install-${PROCESSOR}-minimal"
35  for i in ${GENTOO_LIST}*\.iso
36  do
37  [ -f $i ] && rm ${verb} -f $i
38  done
39  for i in ${GENTOO_LIST}*\.txt*
40  do
41  [ -f $i ] && rm ${verb} -f $i
42  done
43  local downloaded=""
44  local verb=""
45  if ! ${VERBOSE}
46  then
47  verb="-s"
48  else
49  echo "Downloading from: ${MIRROR}/releases/${PROCESSOR}/autobuilds/\
50 latest-install-${PROCESSOR}-minimal.txt"
51  fi
52 
53  if ! curl -L -O "${MIRROR}/releases/${PROCESSOR}/autobuilds/\
54 latest-install-${PROCESSOR}-minimal.txt" ${verb}
55  then
56  ${LOG[*]} "[ERR] Could not download live CD from Gentoo mirror"
57  exit 1
58  fi
59 
60  local current="$(grep install-${PROCESSOR}-minimal.*.iso \
61 latest-install-${PROCESSOR}-minimal.txt | sed -E 's/iso.*$/iso/' )"
62  local downloaded=$(basename "${current}")
63  ${LOG[*]} "[INF] Downloading $current..."
64 
65  if [ -n "${current}" ]
66  then
67  curl -L -O "${MIRROR}/releases/${PROCESSOR}/autobuilds/${current}" \
68  ${verb}
69  curl -L "${MIRROR}/releases/${PROCESSOR}/autobuilds/${current}.DIGESTS.asc" \
70  ${verb} -o checksums_install.txt
71  else
72  ${LOG[*]} "[ERR] Could not download current install iso."
73  exit 1
74  fi
75 
76  [ $? != 0 ] && ${LOG[*]} "[ERR] Could not download live CD" && exit 1
77  if ! "${DISABLE_CHECKSUM}" && [ -f checksums_install.txt ]
78  then
79  sha512=$(sha512sum "${downloaded}" | cut -f 1 -d ' ')
80  sha512_=$(grep "${downloaded}" checksums_install.txt | head -n1 | cut -f 1 -d ' ')
81 
82  if [ -n "${sha512}" ] && [ -n "${sha512_}" ] && [ "${sha512}" != "${sha521_}" ]
83  then
84  ${LOG[*]} "[MSG] Verified SHA512SUM of ${downloaded}."
85  else
86  ${LOG[*]} "[MSG] SHA512SUM of ${downloaded} did not match digest."
87  ${LOG[*]} "[MSG] Computed sha512sum of file: ${sha512}"
88  ${LOG[*]} "[MSG] Digest sha512sum of file: ${sha512_}"
89  exit 1
90  fi
91  rm -f checksums_install.txt
92  fi
93 
94  if [ -f "${downloaded}" ]
95  then
96  verb=""
97  ${VERBOSE} && verb="-v"
98  ${LOG[*]} "[INF] Caching downloaded ISO to ${CACHED_ISO}"
99  cp ${verb} -f ${downloaded} ${CACHED_ISO}
100  mv ${verb} ${downloaded} ${ISO}
101  if [ -f ${ISO} ]
102  then
103  export LIVECD=${ISO}
104  else
105  ${LOG[*]} "[ERR] You need to fetch an install ISO (${ISO}) file!"
106  exit 1
107  fi
108  else
109  ${LOG[*]} "[ERR] Could not find downloaded live CD ${downloaded}"
110  exit 1
111  fi
112 }

119 fetch_clonezilla_iso() {
120 
121  local verb=""

124  <<< ${DOWNLOAD_CLONEZILLA_PATH})"

132  fi

134  if_fails $? "[ERR] Could not download CloneZilla CD."
135

138  if ! curl -L -O "${GITHUB_RELEASE_PATH}/blob/master/SUMS.txt" ${verb}
139  then
140  ${LOG[*]} "Could not download CloneZilla checksums."
141  exit 1
142  else
143  md5="$(grep -o -E 'MD5SUM: ([0-9a-z]+)' SUMS.txt | cut -f 2 -d' ')"
144  md5_=$(md5sum "${clonezilla_iso}"| cut -f 1 -d' ')
145 
146  if [ -n "${md5}" ] && [ -n "${md5_}" ] && [ "${md5}" = "${md5_}" ]
147  then
148  ${LOG[*]} "[MSG] Verified checksum for ${clonezilla_iso}"
149  else
150  ${LOG[*]} "[ERR] Checksum for ${clonezilla_iso} did not match digest:"
151  ${LOG[*]} "[MSG] Computed md5sum of file: ${md5_}"
152  ${LOG[*]} "[MSG] Digest md5sum of file: ${md5}"
153  exit 1
154  fi
155  rm -f SUMS.txt
156  fi
157  fi
158 
159  export CLONEZILLACD="${clonezilla_iso}"
160 
161  # first cache it
162 
163  verb=""
164  "${VERBOSE}" && verb="-v"
165  cp ${verb} -f "${CLONEZILLACD}" clonezilla.iso
166 }

199 ## @fn fetch_process_clonezilla_iso()
200 ## @brief Download clonezilla ISO file and process it.
201 ## @details
202 ## @li Download iff #DOWNLOAD_CLONEZILLA has value "true"
203 ## @li Optionnally checks MD5 sum.
204 ## @li Caches ISO download for later uses.
205 ## @li Uncaches it if no fresh download.
206 ## @li Mount ISO download. Copy ro mounted filesystem to rw directory.
207 ## @li Unsquash ISO filesystem.squashfs.
208 ## @li Copy clonezilla config file.
209 ## @li Copy resolv.conf to unsquashed filesystem.
210 ## @retval 0 on success or exits -1 on failure.
211 ## @ingroup fetchFunctions
212 
213 fetch_process_clonezilla_iso() {
214 
215  cache_uncache_clonezilla_iso
216  cd "${VMPATH}" || exit 2
217  local verb=""
218  "${VERBOSE}" && verb=-v
219 
220  # now cleanup, mount and copy CloneZilla live CD
221 
222  if [ ! -d mnt ]
223  then
224  mkdir mnt
225  else
226  if mountpoint mnt
227  then
228  if ! umount -l mnt
229  then
230  ${LOG[*]} "[ERR] Could not unmount mnt."
231  ${LOG[*]} "[ERR] Fatal. Exiting..."
232  exit 4
233  fi
234  fi
235  rm ${verb} -rf mnt && mkdir mnt || exit 2
236  fi
237 
238  [ ! -d mnt2 ] && mkdir mnt2 || { rm ${verb} -rf mnt2 && mkdir mnt2; }
239 
240  if "${USE_BSDTAR}"
241  then
242  cd mnt2
243  BSDTAR_BINARY="$(which bsdtar)"
244  if_fails $? "[ERR] Could not find bsdtar."
245  "${BSDTAR_BINARY}" xpf ../"${CLONEZILLACD}"
246  if_fails $? "[ERR] Could not extract ${CLONEZILLACD} to mnt2 using bsdtar."
247  cd -
248  else
249  "${VERBOSE}" && ${LOG[*]} "[INF] Mounting CloneZilla CD ${CLONEZILLACD}"
250  mount -oloop "${CLONEZILLACD}" ./mnt \
251  || { ${LOG[*]} "[ERR] Could not mount ${CLONEZILLACD} to mnt"
252  exit 1; }
253  "${VERBOSE}" \
254  && ${LOG[*]} "[INF] Now syncing CloneZilla CD to mnt2 in rw mode."
255  rsync ${verb} -a ./mnt/ mnt2 \
256  || { ${LOG[*]} "[ERR] Could not copy clonezilla files to mnt2"
257  exit 1; }
258  fi
259 
260  cd mnt2/live || exit 2
261  unsquashfs filesystem.squashfs \
262  || { ${LOG[*]} "[ERR] Failed to unsquash clonezilla's filesystem.squashfs"
263  exit 1; }
264  cp ${verb} --dereference /etc/resolv.conf squashfs-root/etc
265  cd "${VMPATH}"
266  return 0
267 }
