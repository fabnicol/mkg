24 ## @fn fetch_gentoo_install_iso()
25 ## @brief Download minimal Gentoo install ISO file and caches.q
26 ## @details Optionaly checks MD5SUMS.
27 ## @ingroup fetchFunctions
28 
29 fetch_gentoo_install_iso() {
30 
31  local verb="-v"
32  ! ${VERBOSE} && verb=""
33 
34  local GENTOO_LIST="install-${PROCESSOR}-minimal"
35  for i in ${GENTOO_LIST}*\.iso
36  do
37  [ -f $i ] && rm ${verb} -f $i
38  done
39  for i in ${GENTOO_LIST}*\.txt*
40  do
41  [ -f $i ] && rm ${verb} -f $i
42  done
43  local downloaded=""
44  local verb=""
45  if ! ${VERBOSE}
46  then
47  verb="-s"
48  else
49  echo "Downloading from: ${MIRROR}/releases/${PROCESSOR}/autobuilds/\
50 latest-install-${PROCESSOR}-minimal.txt"
51  fi
52 
53  if ! curl -L -O "${MIRROR}/releases/${PROCESSOR}/autobuilds/\
54 latest-install-${PROCESSOR}-minimal.txt" ${verb}
55  then
56  ${LOG[*]} "[ERR] Could not download live CD from Gentoo mirror"
57  exit 1
58  fi
59 
60  local current="$(grep install-${PROCESSOR}-minimal.*.iso \
61 latest-install-${PROCESSOR}-minimal.txt | sed -E 's/iso.*$/iso/' )"
62  local downloaded=$(basename "${current}")
63  ${LOG[*]} "[INF] Downloading $current..."
64 
65  if [ -n "${current}" ]
66  then
67  curl -L -O "${MIRROR}/releases/${PROCESSOR}/autobuilds/${current}" \
68  ${verb}
69  else
70  ${LOG[*]} "[ERR] Could not download current install iso."
71  exit 1
72  fi
73 
74  [ $? != 0 ] && ${LOG[*]} "[ERR] Could not download live CD" && exit 1
75  ! "${DISABLE_MD5_CHECK}" && check_md5sum "${downloaded}"
76  if [ -f "${downloaded}" ]
77  then
78  verb=""
79  ${VERBOSE} && verb="-v"
80  ${LOG[*]} "[INF] Caching downloaded ISO to ${CACHED_ISO}"
81  cp ${verb} -f ${downloaded} ${CACHED_ISO}
82  mv ${verb} ${downloaded} ${ISO}
83  if [ -f ${ISO} ]
84  then
85  export LIVECD=${ISO}
86  else
87  ${LOG[*]} "[ERR] You need to fetch an install ISO (${ISO}) file!"
88  exit 1
89  fi
90  else
91  ${LOG[*]} "[ERR] Could not find downloaded live CD ${downloaded}"
92  exit 1
93  fi
94 }

101 fetch_clonezilla_iso() {
102 
103  local verb=""
104  ${LOG[*]} "[INF] Downloading CloneZilla..."
105  local clonezilla_file="$(sed -E 's/.*\/(.*)\/download/\1/' \
106  <<< ${DOWNLOAD_CLONEZILLA_PATH})"
107  ! "${VERBOSE}" && verb="-s"
108  if ! curl -L "${DOWNLOAD_CLONEZILLA_PATH}" -o "${clonezilla_file}" ${verb}
109  then
110  ${LOG[*]} "Could not download CloneZilla iso from ${DOWNLOAD_CLONEZILLA_PATH}"
111  exit 1
112  fi
113  local clonezilla_iso="$(ls clonezilla-live*${PROCESSOR}.iso)"
114  [ ${DISABLE_MD5_CHECK} = "false" ] && check_md5sum "${clonezilla_iso}"
115  export CLONEZILLACD="${clonezilla_iso}"
116 
117  # first cache it
118 
119  verb=""
120  "${VERBOSE}" && verb="-v"
121  cp ${verb} -f "${CLONEZILLACD}" clonezilla.iso
122 }

125 ## @brief Either download clonezilla ISO file and caches it OR uncaches it.

128 cache_uncache_clonezilla_iso() {
129  if "${DOWNLOAD_CLONEZILLA}" && ! "${CLONEZILLA_INSTALL}"
130  then
131 
132  # If CLONEZILLA_INSTALL == true the ISO has already been downloaded
133 
134  fetch_clonezilla_iso
135  else

137  then
138 
139  # uncache
140

142  CLONEZILLACD=clonezilla.copy.iso
143  if ! cp -vf clonezilla.iso ${CLONEZILLACD}; then
144  ${LOG[*]} "[ERR] Could not uncache clonezilla.iso"
145  exit 1
146  fi
147  else
148  ${LOG[*]} "[WAR] No CloneZilla ISO file was found.\
149  Please run again with 'download_clonezilla=true'"
150  exit 1
151  fi
152  fi
153 }

155 ## @fn fetch_process_clonezilla_iso()
156 ## @brief Download clonezilla ISO file and process it.
157 ## @details
158 ## @li Download iff #DOWNLOAD_CLONEZILLA has value "true"
159 ## @li Optionnally checks MD5 sum.
160 ## @li Caches ISO download for later uses.
161 ## @li Uncaches it if no fresh download.
162 ## @li Mount ISO download. Copy ro mounted filesystem to rw directory.
163 ## @li Unsquash ISO filesystem.squashfs.
164 ## @li Copy clonezilla config file.
165 ## @li Copy resolv.conf to unsquashed filesystem.
166 ## @retval 0 on success or exits -1 on failure.
167 ## @ingroup fetchFunctions
168 
169 fetch_process_clonezilla_iso() {
170 
171  cache_uncache_clonezilla_iso
172  cd "${VMPATH}" || exit 2
173  local verb=""
174  "${VERBOSE}" && verb=-v
175 
176  # now cleanup, mount and copy CloneZilla live CD
177 
178  if [ ! -d mnt ]
179  then
180  mkdir mnt
181  else
182  if mountpoint mnt
183  then
184  if ! umount -l mnt
185  then
186  ${LOG[*]} "[ERR] Could not unmount mnt."
187  ${LOG[*]} "[ERR] Fatal. Exiting..."
188  exit 4
189  fi
190  fi
191  rm ${verb} -rf mnt && mkdir mnt || exit 2
192  fi
193 
194  [ ! -d mnt2 ] && mkdir mnt2 || { rm ${verb} -rf mnt2 && mkdir mnt2; }
195 
196  "${VERBOSE}" && ${LOG[*]} "[INF] Mounting CloneZilla CD ${CLONEZILLACD}"
197  mount -oloop "${CLONEZILLACD}" ./mnt \
198  || { ${LOG[*]} "[ERR] Could not mount ${CLONEZILLACD} to mnt"
199  exit 1; }
200  "${VERBOSE}" \
201  && ${LOG[*]} "[INF] Now syncing CloneZilla CD to mnt2 in rw mode."
202  rsync ${verb} -a ./mnt/ mnt2 \
203  || { ${LOG[*]} "[ERR] Could not copy clonezilla files to mnt2"
204  exit 1; }
205  cd mnt2/live || exit 2
206  unsquashfs filesystem.squashfs \
207  || { ${LOG[*]} "[ERR] Failed to unsquash clonezilla's filesystem.squashfs"
208  exit 1; }
209  cp ${verb} --dereference /etc/resolv.conf squashfs-root/etc
210  cd "${VMPATH}"
211  return 0
212 }

215 ## @brief Download automatic output of preprocessed_gentoo_install.iso from

220 fetch_preprocessed_gentoo_install() {
221 
222 local verb=""
223 ! "${VERBOSE}" && verb="-s"
224 sleep 3
225 ${LOG[*]} "[INF] Downloading Gentoo minimal install ISO updated with MKG scripts \
226 from Github Actions..."
227 sleep 3
228 ${LOG[*]} <<< "$(curl -L ${GITHUB_RELEASE_PATH2}/${WORKFLOW_TAG2}/\
229 downloaded.iso -o preprocessed_gentoo_install.iso ${verb} 2>&1 | xargs echo '[INF]')"
230 
231 if_fails $? "[ERR] Could not download preprocessed Gentoo install ISO from URL \
232 ${GITHUB_RELEASE_PATH2}/${WORKFLOW_TAG2}"
233 
234 ${LOG[*]} <<< "$(curl -L -O ${GITHUB_RELEASE_PATH2}/${WORKFLOW_TAG2}/\
235 checksums.txt ${verb} 2>&1 | xargs echo '[INF]')"
236 
237 if_fails $? "[ERR] Could not download checksums.txt from URL \
238 ${GITHUB_RELEASE_PATH2}/${WORKFLOW_TAG2}"
239 
240 if ! ${DISABLE_MD5_CHECK}
241 then
242  local md5=$(md5sum "preprocessed_gentoo_install.iso" | cut -f 1 -d' ')
243  [ -f checksums.txt ] && rm -f checksums.txt
244  local md5_=cat 'checksums.txt' | xargs | cut -f2 -d' '
245  if [ ${md5} != ${md5_} ]
246  then
247  ${LOG[*]} "[ERR] MD5 sum of preprocessed_gentoo_install.iso from Github Actions \
248 could not be checked against downloaded file."
249  exit 2
250  fi
251 fi
252 
253 }

295 ## @fn fetch_livecd()
296 ## @brief Downloads Gentoo install CD
297 ## @details Caches it as ${ISO}
298 ## @retval Returns 0 on success or -1 on exit
299 ## @ingroup fetchFunctions
300 
301 fetch_livecd() {
302  cd "${VMPATH}"
303  local CACHED_ISO="install-${PROCESSOR}-minimal.iso"
304 
305  # Use clonezilla ISO for headless VM and Gentoo minimal install ISO for gui
306  # VM
307 
308  if "${CLONEZILLA_INSTALL}"
309  then
310  DOWNLOAD_CLONEZILLA="${DOWNLOAD}"
311  CACHED_ISO=clonezilla.iso
312  else
313  "${DOWNLOAD}" && fetch_gentoo_install_iso
314  fi
315  if "${DOWNLOAD_CLONEZILLA}"
316  then
317  fetch_clonezilla_iso
318  "${CLONEZILLA_INSTALL}" && ISO="${CLONEZILLACD}"
319  else
320  if ! ${USE_CLONEZILLA_WORKFLOW}
321  then
322  [ -f "clonezilla.iso" ] && CLONEZILLACD="clonezilla.iso" \
323  || { ${LOG[*]} "[ERR] CloneZilla ISO has not been cached. \
324 Run with download=true" ; exit 1; }
325  fi
326  fi
327  if ! "${DOWNLOAD}"
328  then
329  if "${CREATE_SQUASHFS}"
330  then
331  [ -f ${CACHED_ISO} ] \
332  && { ${LOG[*]} "[INF] Uncaching ${ISO} from ${CACHED_ISO}"
333  cp -f ${CACHED_ISO} "${ISO}"; }
334  else
335  ${LOG[*]} "[ERR] No minimal install Gentoo ISO was found, \
336 please rerun with download=true"
337  exit 1
338  fi
339  LIVECD="${ISO}"
340  fi
341  return 0
342 }
