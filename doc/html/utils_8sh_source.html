32 ## @fn create_options_array()
33 ## @brief Read file @b options into a temporary array A
34 ## which will contain command line specifications
35 ## @details Later on A initializes the read-only array #ARR
36 ## @ingroup auxiliaryFunctions
37 
38 ## @var ARR
39 ## @brief global string array of switches and default values
40 ## @details Structure is as follows: @code
41 ## {{"Commandline option", "Description", "Default value", "Type"}, {...},...}
42 ## @endcode
43 ## 'Type' is among the following values:
44 ## @li @b b Boolean, 'false' or 'true'
45 ## @li @b d An existing directory
46 ## @li @b e Email address: regexp "[a-z]+@[a-z]+\.[a-z]+"
47 ## @li @b f An existing file
48 ## @li @b n Numeric value
49 ## @li @b o 'on' or 'off', a VBoxManage custom Boolean
50 ## @li @b s Non-empty string. Corresponding defaults may be empty however.
51 ## This is the notably case for passwords. For such options, explicit
52 ## commandline value after '=' is requested.
53 ## @li @b u Url
54 ## @li @b x:y Conditional type x: one of the above, with [ -z "$x" ]
55 ## <=> { [ "$y" = "false" ] || [ -z "$y" ]; } && [ "$y" != "true" ]
56 ## @li @b vm Restricted to the <tt>vm</tt> option:
57 ## can be set to @ code <tt>false</tt> to bypass OS building and only
58 ## perform burning/external device operations.
59 ## A double-entry array will be simulated using indexes.
60 ## @ingroup createInstaller
61 ## @note `debug_mode` should be place up front in the array
62 
63 declare -a ARR
64 
65 ## @var ARRAY_LENGTH
66 ## @brief Number of switches (true length of array divided by 4)
67 ## @ingroup createInstaller
68 
69 declare -i ARRAY_LENGTH
70 
71 create_options_array() {
72 
73  # language note: IFS is the bash array separator, not awk's
74 
75  IFS=';'
76  read -r -a A <<< $(awk -F"\"" \
77  '/^\s*$/ {next;} {if ( ! match($1, "#")) printf "%s;%s;%s;%s;",$2,$4,$6,$8}' \
78  "$1")
79  ARR=(${A[@]})
80  ARRAY_LENGTH=$((${#ARR[*]}/4))
81  export ARR
82  export ARRAY_LENGTH
83  unset A
84  IFS=" "
85 }

197 ## @fn get_device()
198 ## @param Either a device KNAME (sdX), or a directory mountpoint
199 ## or enough letters of the VENDOR for grep to uniquely
200 ## identify it in @code lsblk ouput.
201 ## @brief Give device from mount folder input.
202 ## @ingroup auxiliaryFunctions
203 
204 get_device() {
205  if is_block_device "$1"
206  then
207  echo "$1"
208  else
209  local res
210  if [ -d "$1" ]
211  then
212  res=$(findmnt --raw --first -a -n -c "$1" | cut -f2 -d' ')
213  res=$(sed -r 's/\/dev\/([a-zA-Z]+)[0-9]+/\1/' <<< "${res}")
214  echo "${res}"
215  else
216  res=$(find_device_by_vendor "$1")
217  if [ -n "${res}" ]
218  then
219  if "${INTERACTIVE}"
220  then
221  read -p "[WAR] Please confirm there is no error and you wish \
222 to create an installer with device /dev/${res}, vendor name $1. Reply with \
223 uppercase Y to continue: " reply
224  if [ "${reply}" = "Y" ]
225  then
226  get_device "${res}"
227  else
228  exit 0
229  fi
230  fi
231  else
232  ${LOG[*]} "[ERR] $1 is neither a mountpoint nor a \
233 block device"
234  exit 1
235  fi
236  fi
237  fi
238 }

304 ## @fn burn_iso()
305 ## @brief Burn Gentoo clonezilla installer to DVD using \b cdrecord
306 ## @note \b cdrecord should have been installed beforehand
307 ## @retval Return \b cdrecord return code
308 ## @ingroup auxiliaryFunctions
309 
310 burn_iso() {
311  if [ -z "${CDRECORD}" ]
312  then
313  if ! which cdrecord
314  then
315  ${LOG[*]} "[ERR] Could not find cdrecord"
316  ${LOG[*]} "[ERR] Please install cdrtools in your PATH or \
317  specify cdrecord full filepath on commandline:"
318  ${LOG[*]} " burn=true \
319 cdrecord=/path/to/cdrecord/executable"
320  exit 1
321  else
322  CDRECORD="$(which cdrecord)"
323  test_cdrecord
324  fi
325  else
326  test_cdrecord
327  fi
328 
329  "${BLANK}" && BLANK='blank=fast' || BLANK=""
330 
331  if [ -z "${SCSI_ADDRESS}" ] || [ "${SCSI_ADDRESS}" = "dep" ]
332  then
333  OPT=("-eject" "${BLANK}")
334  else
335  OPT=("-eject" "${BLANK}" "dev=${SCSI_ADDRESS}")
336  fi
337 
338  if [ ! -f "${ISO_OUTPUT}" ]
339  then
340  ${LOG[*]} "[ERR] No such files as ${ISO_OUTPUT}"
341  exit 1
342  fi
343  ${LOG[*]} "[INF] Burning installation medium ${ISO_OUTPUT} \
344 to optical disc..."
345  "${VERBOSE}" && ${LOG[*]} "[MSG] ${CDRECORD} ${OPT[*]} ${ISO_OUTPUT}"
346 
347  # do not quote OPT
348 
349  ${CDRECORD} ${OPT[*]} "${ISO_OUTPUT}"
350  if_fails $? "[ERR] Could not burn ${ISO_OUTPUT}"
351 }
